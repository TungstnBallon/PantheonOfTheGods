# called for the newborn child
# scope:child is the new born character
# scope:mother is the pregnant mother
# scope:father is the family father (not the biological)
# scope:real_father is the biological father
# scope:is_bastard, true if a known bastard
on_birth_child = {
	random_events = {
		10 = pog_changeling_birth.0001
	}

	on_actions = {
		on_deity_birth_gamerules
		on_deity_birth
		on_human_birth
		on_dwezgaz_birth
		on_elf_birth
	}
}

on_elf_birth = {
	effect = {
		# ELves Assignment
		if = {
			limit = {
				AND = {
					scope:real_father = {
						is_elf = yes
					}
					scope:mother = {
						is_elf = yes
					}
				}
			}
			add_trait = elf
		}

		if = {
			limit = {
				AND = {
					scope:real_father = {
						is_half_elf = yes
					}
					scope:mother = {
						is_half_elf = yes
					}
				}
			}
			add_trait = half_elf
		}

		if = {
			limit = {
				OR = {
					AND = {
						scope:real_father = {
							is_elf = no
							is_half_elf = no
						}
						scope:mother = {
							is_elf = yes
							is_half_elf = no
						}
					}
					AND = {
						scope:real_father = {
							is_elf = yes
							is_half_elf = no
						}
						scope:mother = {
							is_elf = no
							is_half_elf = no
						}
					}
				}
			}
			add_trait = half_elf
		}

		if = {
			limit = {
				OR = {
					AND = {
						scope:real_father = {
							is_half_elf = yes
						}
						scope:mother = {
							is_elf = yes
							is_half_elf = no
						}
					}
					AND = {
						scope:real_father = {
							is_elf = yes
							is_half_elf = no
						}
						scope:mother = {
							is_half_elf = yes
						}
					}
				}
			}
			random_list = {
				40 = {
					add_trait = elf
				}
				60 = {
					add_trait = half_elf
				}
			}
		}

		if = {
			limit = {
				OR = {
					AND = {
						scope:real_father = {
							is_half_elf = yes
						}
						scope:mother = {
							is_half_elf = no
							is_elf = no
						}
					}
					AND = {
						scope:real_father = {
							is_half_elf = no
							is_elf = no
						}
						scope:mother = {
							is_half_elf = yes
						}
					}
				}
			}
			random_list = {
				40 = {
					add_trait = human
				}
				60 = {
					add_trait = half_elf
				}
			}
		}
	}
}

on_deity_birth = {
	effect = {
		if = {
			limit = {
				AND = {
					scope:real_father = {
						is_pure = yes
					}
					scope:mother = {
						is_pure = yes
					}
				}
			}
			add_trait = young_pure_deity
		}

		if = {
			limit = {
				AND = {
					scope:real_father = {
						is_fallen = yes
					}
					scope:mother = {
						is_fallen = yes
					}
				}
			}
			add_trait = young_fallen_deity
		}

		if = {
			limit = {
				AND = {
					scope:real_father = {
						has_trait = demi_deity
					}
					scope:mother = {
						has_trait = demi_deity
					}
				}
			}
			add_trait = demi_deity
		}

		if = {
			limit = {
				OR = {
					AND = {
						scope:real_father = {
							AND = {
								NOT = { has_trait = demi_deity }
								is_pure = no
								is_fallen = no
							}
						}
						scope:mother = {
							has_trait = demi_deity
						}
					}
					AND = {
						scope:real_father = {
							has_trait = demi_deity
						}
						scope:mother = {
							AND = {
								NOT = { has_trait = demi_deity }
								is_pure = no
								is_fallen = no
							}
						}
					}
				}
			}
			add_trait = blood_of_gods
		}

		if = {
			limit = {
				OR = {
					AND = {
						scope:real_father = {
							is_pure = yes
						}
						scope:mother = {
							has_trait = demi_deity
						}
					}
					AND = {
						scope:real_father = {
							has_trait = demi_deity
						}
						scope:mother = {
							is_pure = yes
						}
					}
				}
			}
			random_list = {
				50 = {
					add_trait = demi_deity
				}
				50 = {
					add_trait = young_pure_deity
				}
			}
		}

		if = {
			limit = {
				OR = {
					AND = {
						scope:real_father = {
							is_fallen = yes
						}
						scope:mother = {
							has_trait = demi_deity
						}
					}
					AND = {
						scope:real_father = {
							has_trait = demi_deity
						}
						scope:mother = {
							is_fallen = yes
						}
					}
				}
			}
			random_list = {
				50 = {
					add_trait = demi_deity
				}
				50 = {
					add_trait = young_fallen_deity
				}
			}
		}

		if = {
			limit = {
				OR = {
					AND = {
						scope:real_father = {
							AND = {
								NOT = { has_trait = demi_deity }
								is_pure = no
								is_fallen = no
							}
						}
						scope:mother = {
							is_fallen = yes
						}
					}
					AND = {
						scope:real_father = {
							is_fallen = yes
						}
						scope:mother = {
							AND = {
								NOT = { has_trait = demi_deity }
								is_pure = no
								is_fallen = no
							}
						}
					}
				}
			}
			add_trait = demi_deity
		}

		if = {
			limit = {
				OR = {
					AND = {
						scope:real_father = {
							AND = {
								NOT = { has_trait = demi_deity }
								is_pure = no
								is_fallen = no
							}
						}
						scope:mother = {
							is_pure = yes
						}
					}
					AND = {
						scope:real_father = {
							is_pure = yes
						}
						scope:mother = {
							AND = {
								NOT = { has_trait = demi_deity }
								is_pure = no
								is_fallen = no
							}
						}
					}
				}
			}
			add_trait = demi_deity
		}

		if = {
			limit = {
				OR = {
					AND = {
						scope:real_father = {
							is_pure = yes
						}
						scope:mother = {
							is_fallen = yes
						}
					}
					AND = {
						scope:real_father = {
							is_fallen = yes
						}
						scope:mother = {
							is_pure = yes
						}
					}
				}
			}
			random_list = {
				50 = {
					add_trait = young_fallen_deity
				}
				50 = {
					add_trait = young_pure_deity
				}
			}
		}
	}
}

on_pregnancy_mother = {
	on_actions = {
		on_blood_dilution
	}
}

on_blood_dilution = {
	trigger = {
		has_game_rule = no_blood_dilution_type
	}

	effect = {
		if = {
			limit = {
				OR = {
					AND = {
						scope:real_father = {
							is_deity = yes
						}
						scope:mother = {
							is_deity = no
						}
					}
					AND = {
						scope:real_father = {
							is_deity = no
						}
						scope:mother = {
							is_deity = yes
						}
					}
				}
			}
			remove_trait = pregnant
			end_pregnancy = yes
			pregnancy_cleanup_effect = yes
			add_character_modifier = {
				modifier = attempted_divine_pregnancy
				years = 1
			}
		}
	}
}
